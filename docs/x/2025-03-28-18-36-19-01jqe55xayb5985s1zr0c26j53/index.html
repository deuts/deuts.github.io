To suppress the #delegation warning in #PowerApps while using #Sharepoint as the datasource, here’s the formula that worked when using the Search function:

    With(
        { MyCollection:
            Filter('Bank Transactions',
                !\*chkBankCode\_1\*.Checked || StartsWith(MPIRefID,\*cmbBankCode\_1\*.Selected.Title),
                !\*chkDateFrom\_1\*.Checked || ((Date >= \*dpDateFrom\_1\*.SelectedDate) && (Date <= \*dpDateTo\_1\*.SelectedDate)),
                !\*chkAmtFrom\_1\*.Checked || ((Amount >= Value(\*txtInputAmtFrom\_1\*.Value)) && (Amount <= Value(\*txtInputAmtTo\_1\*.Value))),
                !\*chkCancelled\_1\*.Checked || (Status = Blank()),
                !\*chkCleared\_1\*.Checked || (Closed = false) || (Closed = Blank()),
                !\*chkDebits\_1\*.Checked || (Amount > 0),
                !\*chkTnxType\_1\*.Checked || (TnxType = \*cmbTnxType\_1\*.Selected.ID)
            )
        },
        ClearCollect(BankTnxSearchColl,Search(MyCollection,\*txtInputOtherInfo\_1\*.Value,'Other Info')
    );
